/*Создайте представление, которое выводит название name товарной позиции из таблицы products и 
соответствующее название каталога name из таблицы catalogs./*

CREATE OR REPLACE VIEW products_catalogs AS
SELECT
p.name AS product,
c.name AS catalogs
FROM products AS p
INNER JOIN
catalogs AS c 
ON 
p.catalog_id = c.id;


“Хранимые процедуры и функции, триггеры"
/*Создайте хранимую функцию hello(), которая будет возвращать приветствие, в зависимости от текущего времени суток. 
С 6:00 до 12:00 функция должна возвращать фразу "Доброе утро", с 12:00 до 18:00 функция должна возвращать фразу "Добрый день", 
с 18:00 до 00:00 — "Добрый вечер", с 00:00 до 6:00 — "Доброй ночи"./*

CREATE FUNCTION hello ()
RETURNS TINYTEXT NO SQL --указываем тип возвращаемого значения
BEGIN
DECLARE hour INT; ---объявляем переменную
SET hour = HOUR(NOW()); --наполняем переменную текущим временем
CASE --в зависимости от значения выводим результат
WNEN hour BETWEEN 0 AND 5 THEN
RETURN "Доброй ночи";
WNEN hour BETWEEN 6 AND 11 THEN
RETURN "Доброу утро";
WNEN hour BETWEEN 12 AND 17 THEN
RETURN "Добрый день";
WNEN hour BETWEEN 18 AND 23 THEN
RETURN "Добрый вечер";
END CASE;

DECLIMER ;
SELECT NOW(), hello ();

При попытке присвоить полям NULL-значение необходимо отменить операцию./*

FOR EACH ROW BEGIN --для каждой строки
